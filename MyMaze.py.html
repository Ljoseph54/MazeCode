from pygame.locals import *
import pygame
from random import *

walls = []
rand_color = False


class Player():

    def __init__(self):
        self.rect = pygame.Rect(80, 80, 20, 20)

    # move
    def move(self, xdir, ydir):

        if xdir != 0:
            self.move_in_direction(xdir, 0)
        if ydir != 0:
            self.move_in_direction(0, ydir)

    # move in a direction
    def move_in_direction(self, xdir, ydir):
        global rand_color

        self.rect.x += xdir
        self.rect.y += ydir
        
        #check collisions
        for wall in walls:
            if self.rect.colliderect(wall.rect):
                if xdir > 0:
                    self.rect.right = wall.rect.left
                    self.rect.x = 80
                    self.rect.y = 80
                if xdir < 0:
                    self.rect.left = wall.rect.right
                    self.rect.x = 80
                    self.rect.y = 80
                if ydir > 0:
                    self.rect.bottom = wall.rect.top
                    self.rect.x = 80
                    self.rect.y = 80
                if ydir < 0:
                    self.rect.top = wall.rect.bottom
                    self.rect.x = 80
                    self.rect.y = 80

# initialize the window
pygame.init()
window = pygame.display.set_mode((1920, 1080), FULLSCREEN)
end_rect = pygame.Rect(1450, 970, 40, 40)

player = Player()
maze = Maze()
maze.draw()

# run the game
running = True
while running:

    for i in pygame.event.get():
        if i.type == pygame.QUIT:
            running = False
        if i.type == pygame.KEYDOWN and i.key == pygame.K_ESCAPE:
            running = False

    # key presses
    key = pygame.key.get_pressed()
    if key[pygame.K_LEFT]:
        player.move(-2, 0)
    if key[pygame.K_RIGHT]:
        player.move(2, 0)
    if key[pygame.K_UP]:
        player.move(0, -2)
    if key[pygame.K_DOWN]:
        player.move(0, 2)
    if pygame.mouse.get_pressed() == (1, 0, 0):
        rand_color = True
    if pygame.mouse.get_pressed() == (0, 0, 1):
        rand_color = False

    # if rand color is off
    if rand_color == False:
        window.fill((0, 0, 0))
        # draw walls
        for wall in walls:
            pygame.draw.rect(window, (255, 255, 255), wall.rect)
        pygame.draw.rect(window, (0, 255, 255), player.rect)
        pygame.draw.rect(window, (0, 255, 0), end_rect)

    # if rand color is on
    if rand_color == True:
        for i in range(0, 16):
            for j in range(0, 9):
                window.fill((randint(0, 255), randint(0, 255), randint(0, 255)), (i * 120, j * 120, 120, 120))
        # draw walls
        for wall in walls:
            pygame.draw.rect(window, (randint(0, 255), randint(0, 255), randint(0, 255)), wall.rect)
        pygame.draw.rect(window, (randint(0, 255), randint(0, 255), randint(0, 255)), player.rect)
        pygame.draw.rect(window, (randint(0, 255), randint(0, 255), randint(0, 255)), end_rect)

    # if it gets to the end
    if player.rect.colliderect(end_rect):
        player.rect.x = 80
        player.rect.y = 80

    # updates the screen
    pygame.display.flip()
